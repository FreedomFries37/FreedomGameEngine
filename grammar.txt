#UNIVERSAL TYPES

$PATH = the path of the file
$TYPE

optional rule <string>:
	<charbasic><stringEx>
	
rule <charbasic>:
	[a-z,A-Z]
	
rule <stringEx>:
	<charEx><stringtail>
	
optional rule <stringtail>:
	<stringEx>

rule <charEx>:
	[a-z,A-Z,0-9]
	
optional rule <whitespace>:
	\t<whitespace>
	\n<whitespace>
	 <whitespace>

rule <type>:
	<string>
	 
	 
	 
//For Game Objects

rule <asset>:
	[ASSET]$PATH

rule <gameobject>:
	name="<string>"<whitespace><fieldlist><whitespace>children={<whitespace><childrenlist><whitespace>}<whitespace>components={<whitespace><componentlist><whitespace>}



	
#Fields
optional rule <fieldlist>:
	[<type>]<string>=<fieldlisttail>
	
rule <fieldlisttail>:
	<value><whitespace><fieldlist>
	{<fieldlist>}<whitespace><fieldlist>
	
#Children
optional rule <childrenlist>:
	<gameobject><childrenlist>
	
#Components
optional rule <componentlist>:
	<component>{<whitespace><fieldlist><whitespace>}
	
rule <component>:
	<string>
